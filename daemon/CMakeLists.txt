    SET(CMAKE_REQUIRED_LIBRARIES ${CMAKE_REQUIRED_LIBRARIES} numa)
    CHECK_C_SOURCE_COMPILES("
         #include <numa.h>
         int main() {
            numa_available();
         }" HAVE_LIBNUMA)
    CMAKE_POP_CHECK_STATE()
ENDIF ()

    SET(NUMA_LIBRARIES numa)
    add_definitions(-DHAVE_LIBNUMA=1)
ENDIF ()
